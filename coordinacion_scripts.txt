CREATE TABLE Alumno (
	Cuenta VARCHAR(15) NOT NULL,
	Nombre VARCHAR(40) NOT NULL,
	Carrera VARCHAR(25) NOT NULL,

	Antiguedad VARCHAR(3) NOT NUL,
	Total_cargos VARCHAR(2) NOT NULL,

//PALABRA RESERVADA DISTINCT EN MYSQL 
NOS PERMITE EVITAR DUPLICAR DATOS EN UNA SENTENCIA SELECT 

Por defecto, las consultas SQL muestran todas las filas devueltas, 
incluyendo las duplicadas, en el conjunto de resultados. La palabra 
clave DISTINCT en la cláusula SELECT se utiliza para eliminar las 
filas duplicadas y mostrar una lista única de valores. En otras palabras, 
la palabra clave DISTINCT recupera valores únicos de una tabla.

LA SIGUIENTE CONSULTA NOS DEVUELVE LAS SECCIONES DUPLICADAS DE LA TABLA DOCENTE 
SELECT Seccion FROM Docente;
APLICAMOS LA PALABRA DISTINCT ANTES DE LA PALABRA SELECT
SELECT DISTINCT Seccion FROM Docente;
SELECT DISTINCT Opcion_titulacion FROM Alumno;
